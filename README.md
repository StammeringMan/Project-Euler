Problem 1: Multiples of 3 and 5

function sum(){
  var n = [];
  var total = 0;
  for(var i=1; i<1000; i++){
    if(i % 3 === 0 || i % 5 === 0){
      n.push(i);
      for(var j in n){
        total += n[j];
      } 
    }
  }
  return total;
}  
sum();
